STEP 3: Understand MediaPipe landmark structure
────────────────────────────────────────────────

MediaPipe Hands returns landmarks as objects with properties:
  - landmark[0] = Wrist
  - landmark[1-4] = Thumb (base, middle joint, pip, tip)
  - landmark[5-8] = Index (base, middle joint, pip, tip)
  - landmark[9-12] = Middle (base, middle joint, pip, tip)
  - landmark[13-16] = Ring (base, middle joint, pip, tip)
  - landmark[17-20] = Pinky (base, middle joint, pip, tip)

Key indices:
  - 0 = Wrist (always index 0)
  - 4 = Thumb tip
  - 8 = Index fingertip
  - 12 = Middle fingertip
  - 16 = Ring fingertip
  - 20 = Pinky fingertip

Each landmark has:
  - .x (normalized 0.0-1.0, left-right)
  - .y (normalized 0.0-1.0, top-bottom)
  - .z (normalized depth, -1.0 to 1.0)

def test_cooldown():
    """Test cooldown mechanism prevents double-triggers."""
    detector = GestureDetector()
    
    # TEST CASE 1: Trigger same gesture twice within 100ms
    # Should only return gesture first time, None second time
    hands_data = [
        {'landmarks': create_open_palm()},
    ]
    
    # First call - should detect gesture
    gesture1 = detector.detect_gesture(hands_data)
    assert gesture1 == 'space', "First open palm should be detected"
    print("✓ Test 5.1 passed: First gesture detected")
    
    # Second call immediately after - should be in cooldown
    gesture2 = detector.detect_gesture(hands_data)
    assert gesture2 is None, "Second gesture within cooldown should return None"
    print("✓ Test 5.2 passed: Second gesture blocked by cooldown")
    
    # Wait 100ms more (total 100ms elapsed)
    time.sleep(0.1)
    gesture3 = detector.detect_gesture(hands_data)
    assert gesture3 is None, "Even after 100ms, still in cooldown (500ms total)"
    print("✓ Test 5.3 passed: Still in cooldown after 100ms")
    
    # Wait 400ms more (total 500ms elapsed)
    time.sleep(0.4)
    gesture4 = detector.detect_gesture(hands_data)
    assert gesture4 == 'space', "After 500ms cooldown should expire"
    print("✓ Test 5.4 passed: Gesture detected after cooldown expires")


def test_cooldown_different_gestures():
    """Test that cooldown is per-gesture (not global)."""
    detector = GestureDetector()
    
    # This is optional/advanced test
    # If your implementation tracks cooldown per gesture, different gestures
    # could potentially trigger at same time without blocking each other
    # Current implementation blocks all gestures during cooldown period
    
    hands_open = [{'landmarks': create_open_palm()}]
    hands_fist = [{'landmarks': create_fist()}]
    
    # Open palm detected
    g1 = detector.detect_gesture(hands_open)
    assert g1 == 'space'
    
    # Immediately switch to fist - should be blocked by cooldown
    # (because cooldown is global, not per-gesture)
    g2 = detector.detect_gesture(hands_fist)
    assert g2 is None, "Different gesture still blocked by active cooldown"
    print("✓ Test 5.5 passed: Cooldown blocks all gestures (not per-gesture)")
