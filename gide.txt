PART 5: ADD SignClassifier.predict() METHOD (if not already present)

Check if core/sign_classifier.py has a predict() method. If not, add:

    def predict(self, features):
        """
        Predict gesture from feature vector.
        
        Args:
            features: numpy array of shape (42,) from FeatureExtractor.normalize()
        
        Returns:
            tuple: (label, confidence)
                label: str, gesture class name (e.g., 'A', 'B')
                confidence: float, 0.0-1.0 confidence score
        """
        # Ensure features is right shape
        if features.shape != (42,):
            raise ValueError(f"Expected shape (42,), got {features.shape}")
        
        # Reshape for sklearn (sklearn expects 2D: n_samples, n_features)
        features_2d = features.reshape(1, -1)
        
        # Get prediction and confidence
        pred_index = self.model.predict(features_2d)[0]
        label = self.classes_[pred_index]
        
        # Get confidence (max probability from predict_proba)
        proba = self.model.predict_proba(features_2d)[0]
        confidence = float(np.max(proba))
        
        return label, confidence
